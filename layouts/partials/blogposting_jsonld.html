{{/* BlogPosting JSON-LD ------------------- Emits a schema.org BlogPosting structured-data block for single pages. Coexists with the Article / Review / VideoObject / BreadcrumbList / WebSite JSON-LD blocks emitted from partials/extend_head.html, and with the PaperMod theme's built-in schema block. Rendered only on .Kind == "page" (regular article pages). Safe to invoke from any template — non-page kinds no-op. */}} {{- if eq .Kind "page" -}} {{- $desc := "" -}} {{- with .Description -}} {{- $desc = . -}} {{- else -}} {{- $desc = .Summary | plainify -}} {{- if gt (len $desc) 160 -}} {{- $desc = substr $desc 0 160 -}} {{- end -}} {{- end -}} {{- $img := "" -}} {{- with .Params.cover -}}{{- with .image -}}{{- $img = . -}}{{- end -}}{{- end -}} {{- $absImg := "" -}} {{- if $img -}} {{- if or (hasPrefix $img "http://") (hasPrefix $img "https://") -}} {{- $absImg = $img -}} {{- else -}} {{- $absImg = printf "%s%s" (.Site.BaseURL | strings.TrimRight "/") (printf "/%s" ($img | strings.TrimLeft "/")) -}} {{- end -}} {{- end -}} {{- $authorName := .Site.Params.Author | default "エロナビ編集部" -}} {{- $section := "" -}} {{- with .Params.categories -}}{{- $section = index . 0 -}}{{- end -}} {{- end -}}
